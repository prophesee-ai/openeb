# Create interface lib for internal use only :
add_library(metavision_utils_gtest INTERFACE)
add_library(MetavisionUtils::gtest ALIAS metavision_utils_gtest)
target_include_directories(metavision_utils_gtest
    INTERFACE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/../include>
)

set(gtest_link_libraries 
      GTest::gtest
      GTest::gtest_main
      GTest::gmock
)

target_link_libraries(metavision_utils_gtest
  INTERFACE
      ${gtest_link_libraries}
      Boost::filesystem
      Boost::program_options
)

add_library(metavision_utils_gtest_main INTERFACE)
add_library(MetavisionUtils::gtest-main ALIAS metavision_utils_gtest_main)
target_link_libraries(metavision_utils_gtest_main
  INTERFACE
      MetavisionUtils::gtest
)
